            if(!password||password.length<6){showAuthError('regPassError','Password required (min 6 chars)');shakeAuthInput(document.getElementById('regPassword'));hasError=true}
            if(password!==confirmPass){showAuthError('regConfirmError','Passwords do not match');shakeAuthInput(document.getElementById('regConfirmPass'));hasError=true}
            if(hasError){haptic('error');return}

            isSubmitting=true;haptic('heavy');
            const btn=document.getElementById('registerBtn');
            const txt=document.getElementById('regBtnText');
            txt.innerHTML='<span class="loading-spinner"></span>';btn.disabled=true;

            try{
                // চেক করি username আগে থেকে আছে কিনা
                const q=query(collection(db,"auth_users"),where("username","==",username));
                const existing=await getDocs(q);
                if(!existing.empty){showAuthError('regUserError','This username is already taken');shakeAuthInput(document.getElementById('regUsername'));haptic('error');txt.innerHTML='CREATE ACCOUNT';btn.disabled=false;isSubmitting=false;return}

                const hashedPass=await simpleHash(password);
                const newDoc=doc(collection(db,"auth_users"));
                const uid=newDoc.id;

                await setDoc(newDoc,{
                    username,name,password:hashedPass,
                    createdAt:serverTimestamp(),
                    telegramId:tg?.initDataUnsafe?.user?.id||null
                });

                // users কালেকশনেও এন্ট্রি
                await setDoc(doc(db,"users",uid),{
                    name,balance:0,total_orders:0,
                    createdAt:serverTimestamp()
                },{merge:true});

                // অটো লগইন
                currentUserData={uid,username,name};
                localStorage.setItem('smm_session',JSON.stringify(currentUserData));
                haptic('success');showToast('Account created successfully!','success');
                enterApp(uid,name,username);

            }catch(e){
                console.error('Register error:',e);
                haptic('error');showToast('Registration failed. Try again.','error');
            }finally{txt.innerHTML='CREATE ACCOUNT';btn.disabled=false;isSubmitting=false}
        };

            const btn=document.getElementById('loginBtn');
            const txt=document.getElementById('loginBtnText');
            txt.innerHTML='<span class="loading-spinner"></span>';btn.disabled=true;

            try{
                const q=query(collection(db,"auth_users"),where("username","==",username));

                localStorage.setItem('smm_session',JSON.stringify(currentUserData));
                haptic('success');showToast(`Welcome back, ${userData.name}!`,'success');
                enterApp(userDoc.id,userData.name,userData.username);

            }catch(e){
                console.error('Login error:',e);haptic('error');showToast('Login failed. Try again.','error');
            }finally{txt.innerHTML='LOGIN';btn.disabled=false;isSubmitting=false}

                c.innerHTML+=`<div class="glass-card p-4"><div class="flex justify-between items-start mb-2"><span class="text-[9px] font-mono text-slate-600 bg-slate-800/50 px-2 py-0.5 rounded">${o.apiOrderId||o.id?.slice(-8)}</span><span class="order-status ${st}"><i class="fas ${ic} mr-1 text-[7px]"></i>${o.status||'Pending'}</span></div><h4 class="font-bold text-xs">${o.service||'Service'}</h4><p class="text-[10px] text-slate-500 truncate mt-0.5 font-mono">${o.link||''}</p><div class="dashed-divider my-3"></div><div class="flex justify-between"><div><span class="text-[8px] font-black text-slate-600 uppercase">Qty</span><div class="font-bold text-xs">${Number(o.qty||0).toLocaleString()}</div></div><div><span class="text-[8px] font-black text-slate-600 uppercase">Cost</span><div class="font-bold text-xs text-emerald-400">৳${(o.cost||0).toFixed(2)}</div></div><div class="text-right"><span class="text-[8px] font-black text-slate-600 uppercase">Date</span><div class="text-[10px] text-slate-500">${dateStr}</div><div class="text-[8px] text-slate-600">${timeStr}</div></div></div></div>`;
            });
        }

        function listenUserOrders(uid){
            const q=query(collection(db,"orders"),orderBy("timestamp","desc"));
            onSnapshot(q,(s)=>{
                const l=[];
                s.forEach(d=>{
                    const data = d.data();
                    if(data.uid === uid){ // শুধুমাত্র লগইন করা ইউজারের অর্ডার নিবে
                        l.push({id:d.id,...data});
                    }
                });
                renderOrders(l);
            });
        }
